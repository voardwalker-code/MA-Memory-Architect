//     .integrationMode  — 'nekocore' or 'off'
//     .memorySearch     — function to search memories (optional)
//     .blockedTools     — Set of tool names blocked in current mode
//
//   Returns: array of { tool, result, ok }
// ─────────────────────────────────────────────────────────────────────────────
