function getRoster() {
  try { return JSON.parse(fs.readFileSync(ROSTER_PATH, 'utf8')); }
  catch { return { models: [] }; }
}

function saveRoster(data) {
  fs.writeFileSync(ROSTER_PATH, JSON.stringify(data, null, 2));
}
