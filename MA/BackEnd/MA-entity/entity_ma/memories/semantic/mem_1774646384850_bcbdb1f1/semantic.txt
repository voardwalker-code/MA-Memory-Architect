_clearLongRequestTimer();
    // If no events came through at all, handle gracefully
    if (!gotSteps) hideTyping();

  } catch (e) {
    _clearLongRequestTimer();
    dismissLongRequest();
    hideTyping();
    if (e.name !== 'AbortError') addSystem('Network error: ' + e.message);
  }

  sending = false;
  sendBtn.disabled = false;
  inputEl.focus();
}
