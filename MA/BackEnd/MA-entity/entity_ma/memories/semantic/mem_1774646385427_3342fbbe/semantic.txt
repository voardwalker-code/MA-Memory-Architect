function updateTodoText(index, value) {
  const todos = getTodos();
  if (!todos[index]) return;
  todos[index].text = value;
  todos[index].updatedAt = new Date().toISOString();
  saveTodos(todos);
}

function deleteTodo(index) {
  const todos = getTodos();
  todos.splice(index, 1);
  saveTodos(todos);
  renderTodos();
}
